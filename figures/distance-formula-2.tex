  \begin{axis}[

    axis lines=middle,
    xmin=-1,xmax=8,
    ymin=-1,ymax=7,
    xlabel=\(x\),
    ylabel=\(y\),
    xtick={3,5},xticklabels={\(x_1\),\(x_2\)},
    ytick={2,5},yticklabels={\(y_1\),\(y_2\)}
    ]
